## üîó API Client

```javascript
// services/api.js
import axios from 'axios';

const api = axios.create({
  baseURL: process.env.REACT_APP_API_URL || 'http://localhost:8000',
  timeout: 180000  // 3 minutes pour les requ√™tes LLM
});

export default {
  // Documents
  getDocuments: () => api.get('/api/documents'),
  uploadDocument: (file, patientId) => {
    const formData = new FormData();
    formData.append('file', file);
    formData.append('patient_id', patientId);
    return api.post('/api/documents/upload', formData);
  },
  
  // Q&A
  askQuestion: (question, patientId, documentIds) => 
    api.post('/api/qa/ask', { question, patientId, documentIds }),
  
  // Synth√®se
  generateSynthesis: (documentIds, options) =>
    api.post('/api/synthesis/generate', { documentIds, ...options }),
  
  // Audit
  getAuditLogs: (page, size) => 
    api.get(`/api/audit/logs?page=${page}&size=${size}`),
  getAuditStats: () => api.get('/api/audit/stats')
};
```
